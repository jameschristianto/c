#include <string.h>

using namespace std;

int main()
{
    
    char x[] = "dog.cat.fish";
    
    char *y = strtok(x, ".");
    
    while(y != NULL){
        printf("%s\n", y);
        y = strtok(NULL, ".");
    }
    
    return 0;
}
